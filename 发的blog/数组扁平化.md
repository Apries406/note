# `pop`+`unshift`
```javascript
const arr = [1, [2, [3, [4, [5]]]]]

function flatten(arr) {
	const resArr = []
	const stack = [...arr]
	while (stack.length) {
		const curr = stack.pop()
		console.log(curr)
		if (Array.isArray(curr)) {
			stack.push(...curr) // 把最后一个数组展开再放进去
		} else {
			resArr.unshift(curr) // 添加到数组开头
		}
	}

	return resArr
}

console.log(flatten(arr)) // [1,2,3,4,5]
```
